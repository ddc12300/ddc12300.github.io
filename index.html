Viña Rock

El festival Viña Rock es un reconocido evento musical que, desde su inicio en 1996, ha ido consolidándose como uno de los principales referentes de la escena alternativa en España. A lo largo de su historia, el festival ha buscado proporcionar a su audiencia una experiencia única y memorable en términos de diversidad artística, impulsando a artistas emergentes y establecidos de diversos géneros musicales, como el rock, punk, reggae, hip-hop, metal y música electrónica. El presente trabajo tiene como objetivo analizar la evolución de las playlists del festival Viña Rock en distintos años. Para ello hemos utilizado la API de spotify y python para extraer las playlist oficiales disponibles, de las cuales hemos encontrado desde 2009 hasta su edición más reciente en 2023. A partir de esto, se examinará la cantidad y características de las canciones y artistas incluidos en sus playlists, con el propósito de identificar las tendencias y cambios en el enfoque musical del festival a lo largo del tiempo, y explorar cómo estos aspectos reflejan su crecimiento, diversificación y posicionamiento en el panorama musical nacional e internacional.

Objetivos

Este estudio busca analizar la evolución de las playlists del festival Viña Rock desde 2009 hasta 2023, examinando la cantidad y características de las canciones y artistas incluidos en ellas, con el objetivo de identificar las tendencias y cambios en el enfoque musical del evento a lo largo del tiempo, prestando especial atención a las proporciones de géneros musicales, la diversidad artística y la representación de artistas emergentes y establecidos. A partir de este análisis, el objetivo principal es proponer un cartel para la siguiente edición del festival que refleje los resultados obtenidos y las tendencias musicales actuales, manteniendo la identidad histórica del Viña Rock.

# Importando las bibliotecas necesarias
import time
import csv
import pandas as pd
import spotipy
from spotipy.oauth2 import SpotifyClientCredentials

# Credenciales de la API de Spotify
client_id = 'YOUR_CLIENT_ID'
client_secret = 'YOUR_CLIENT_SECRET'
client_credentials_manager = SpotifyClientCredentials(client_id=client_id, client_secret=client_secret)
sp = spotipy.Spotify(client_credentials_manager=client_credentials_manager)

# Función para obtener todas las canciones de una lista de reproducción
def get_playlist_tracks(playlist_id):
    results = sp.playlist_tracks(playlist_id)
    tracks = results['items']
    # Iterar a través de todas las páginas de la respuesta hasta que no haya más canciones
    while results['next']:
        time.sleep(0.1)
        results = sp.next(results)
        tracks.extend(results['items'])
    return tracks

# Función para extraer información de una canción
def extract_track_info(track, playlist_name):
    time.sleep(0.1)

    # sacar id artista
    artist_id = track['track']['artists'][0]['id']

    # Cantidad de reintentos en caso de error al obtener información del artista
    retries = 3
    for i in range(retries):
        try:
            artist = sp.artist(artist_id)
            break
        except Exception as e:
            # Tiempo de espera antes de reintentar y mostrando un mensaje de error
            sleep_time = (2 ** i) + 1
            print(f"Error al obtener el artista {artist_id}: {e}, reintentando después de {sleep_time} segundos...")
            time.sleep(sleep_time)
    else:
        # Si no fue posible obtener información del artista después de todos los intentos, imprime un mensaje y devuelve None
        print(f"No se pudo obtener información del artista {artist_id} después de {retries} intentos")
        return None

    # Extrayer géneros y número de seguidores del artista
    genres = ', '.join(artist['genres'])
    followers = artist['followers']['total']

    # Obtener la ID de la canción
    song_id = track['track']['id']

    # Definiendo la cantidad de reintentos en caso de error al obtener las características del audio
    retries = 3
    for i in range(retries):
        try:
            audio_features = sp.audio_features(song_id)[0]
            break
        except Exception as e:
            # Determinando el tiempo de espera antes de reintentar y mostrando un mensaje de error
            sleep_time = (2 ** i) + 1
            print(f"Error al obtener las características del audio {song_id}: {e}, reintentando después de {sleep_time} segundos...")
            time.sleep(sleep_time)
    else:
        # Si no es posible obtener las características del audio después de todos los intentos, imprime un mensaje y devuelve None
        print(f"No se pudo obtener las características del audio {song_id} después de {retries} intentos")
        return None

    # Obteniendo la fecha de lanzamiento del álbum y extrayendo el año
    release_date = track['track']['album']['release_date']
    release_year = release_date.split('-')[0]

    # Diccionario con la información de la canción
    track_info = {
        'track_id': track['track']['id'],
        'playlist_name': playlist_name,
        'title': track['track']['name'],
        'artist': track['track']['artists'][0]['name'],
        'artist_id': artist_id,
        'genres': genres,
        'followers': followers,
        'album': track['track']['album']['name'],
        'release_date': release_year,
        'duration_ms': track['track']['duration_ms'],
        'tempo': audio_features['tempo'],
        'energy': audio_features['energy'],
        'danceability': audio_features['danceability'],
        'spotify_url': track['track']['external_urls']['spotify'],
        'popularity': track['track']['popularity']
    }
    return track_info

# Función para guardar el progreso de la extracción de datos en un archivo
def save_progress(playlist_index, track_index):
    with open('progress.txt', 'w') as progress_file:
        progress_file.write(f'{playlist_index},{track_index}')

# Función para cargar el progreso de la extracción de datos desde un archivo
def load_progress():
    try:
        with open('progress.txt', 'r') as progress_file:
            contents = progress_file.read()
            playlist_index, track_index = map(int, contents.split(','))
            return playlist_index, track_index
    except FileNotFoundError:
        return 0, 0

# Escribiendo la cabecera del archivo CSV
csv_header = ['track_id', 'playlist_name', 'title', 'artist', 'artist_id', 'genres', 'followers', 'album', 'release_date', 'duration_ms', 'tempo', 'energy', 'danceability', 'spotify_url', 'popularity']

with open('song_data.csv', 'w', newline='', encoding='utf-8') as csvfile:
    writer = csv.DictWriter(csvfile, fieldnames=csv_header)
    writer.writeheader()

# Lista de IDs de playlist para extraer datos
playlist_ids = ['3zb78wIWvn9X1so8dceWca', '3LVAF8rnqurv9Kzdo8kAsT', '1WacouuToRBMhBlVUVbnzT', '3vsvh15C5NYUXLjBazAVh3', '4vuUpAwE3vTIWf3CTou0wB', '1P0z4BsLOwtrpGEm7wPqFA', '5vcielrvT47NdtHF9VrG9z', '7rT0e4vvH7hHJnFLULVN8m', '6eppRY4vIjAlTCfmtXT4Vy', '0Tx3fUnxs0ZE0O7A8DQozJ']

# Cargar el progreso de extracción de datos desde el archivo
start_playlist_index, start_track_index = load_progress()

# Iterar a través de todas las playlists y extrayendo información de las canciones
for playlist_index, playlist_id in enumerate(playlist_ids[start_playlist_index:], start=start_playlist_index):
    tracks = get_playlist_tracks(playlist_id)
    playlist_name = sp.playlist(playlist_id)['name']

    with open('song_data.csv', 'a', newline='', encoding='utf-8') as csvfile:
        writer = csv.DictWriter(csvfile, fieldnames=csv_header)

        for track_index, track in enumerate(tracks[start_track_index:], start=start_track_index):
            save_progress(playlist_index, track_index)
            track_info = extract_track_info(track, playlist_name)
            if track_info is not None:
                writer.writerow(track_info)
            start_track_index = 0

    print(f"Playlist '{playlist_name}' data saved to song_data.csv.")
    save_progress(playlist_index + 1, 0)

# Función para expandir un DataFrame de artistas con múltiples géneros en filas separadas
def expand_artist_genres(df):
    expanded_artists = []

    for _, row in df.iterrows():
        artist_id = row['artist_id']
        artist = row['artist']
        genres = row['genres']
        followers = row['followers']
        popularity = row['popularity']

        expanded_artists.append({
               'artist_id': artist_id,
               'artist': artist,
               'genres:': genres,
               'followers': followers,
               'popularity': popularity
        })
    return pd.DataFrame(expanded_artists)

# Leer el archivo CSV de datos de canciones y filtrando artistas únicos
df = pd.read_csv('song_data.csv')
unique_artists = df[['artist_id', 'artist', 'genres', 'followers', 'popularity']].drop_duplicates(subset='artist_id')

# DataFrame de artistas con múltiples géneros en filas separadas
sorted_artists_expanded = expand_artist_genres(unique_artists)

# Guardando el DataFrame de artistas ordenados en un archivo CSV
sorted_artists_expanded.to_csv('sorted_artists_expanded.csv', index=False)
